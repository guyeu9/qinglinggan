## 一、问题概述

经过详细排查，发现虽然后端服务代码已实现，但前端界面存在大量"假实现"问题：
- 按钮存在但功能未连接
- 菜单项点击仅显示"功能开发中"提示
- 数据展示但不可交互
- 设置项未持久化保存

**统计**: 18个假实现功能 + 5个部分实现功能

```dart
actions: [
  // 现有按钮...
  
  // 重新分析按钮
  IconButton(
    icon: const Icon(Icons.refresh),
    tooltip: '重新分析',
    onPressed: _isReanalyzing ? null : _reanalyzeIdea,
  ),
  
  // 更多菜单...
],

// 添加重新分析方法
Future<void> _reanalyzeIdea() async {
  setState(() => _isReanalyzing = true);
  
  try {
    await ref.read(ideaDetailProvider(widget.ideaId).notifier).reanalyze();
    
    if (mounted) {
      ScaffoldMessenger.of(context).showSnackBar(
        const SnackBar(content: Text('已重新提交分析任务')),
      );
    }
  } finally {
    if (mounted) {
      setState(() => _isReanalyzing = false);
    }
  }
}
```

```dart
Future<void> _showCategoryDialog() async {
  final categories = await ref.read(categoryRepositoryProvider).getAll();
  
  if (!mounted) return;
  
  final selectedCategoryId = await showDialog<int>(
    context: context,
    builder: (context) => AlertDialog(
      title: const Text('选择分类'),
      content: Column(
        mainAxisSize: MainAxisSize.min,
        children: categories.map((cat) => 
          RadioListTile<int>(
            title: Text('${cat.icon} ${cat.name}'),
            value: cat.id,
            groupValue: _idea?.categoryId,
            onChanged: (value) => Navigator.pop(context, value),
          ),
        ).toList(),
      ),
    ),
  );
  
  if (selectedCategoryId != null && selectedCategoryId != _idea?.categoryId) {
    await ref.read(ideaDetailProvider(widget.ideaId).notifier)
        .updateCategory(selectedCategoryId);
  }
}
```

```dart
Future<void> _logout() async {
  final confirmed = await showDialog<bool>(
    context: context,
    builder: (context) => AlertDialog(
      title: const Text('退出登录'),
      content: const Text('确定要退出吗？这将清除所有本地数据。'),
      actions: [
        TextButton(
          onPressed: () => Navigator.pop(context, false),
          child: const Text('取消'),
        ),
        TextButton(
          onPressed: () => Navigator.pop(context, true),
          style: TextButton.styleFrom(foregroundColor: Colors.red),
          child: const Text('退出'),
        ),
      ],
    ),
  );
  
  if (confirmed == true && mounted) {
    // 清除所有数据
    await IsarDatabase.clear();
    await AIConfig.clearApiKey();
    
    // 重新启动应用
    if (mounted) {
      Navigator.of(context).pushNamedAndRemoveUntil('/', (route) => false);
    }
  }
}
```
